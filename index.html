APIscript

Creating API script

#API script

!/bin/bash

API_KEY="/root/"node_name"-API/API1"

echo -n "Starting "node_name"-node with API key..."
"node_name"-node --start < "$API_KEY"
echo ""node_name"-node started"

# Print the status of **-node
echo "Checking **-node status..."
**-node --status

#setting up crontab to set up a time trigger for scripts

#Crontab

# Run start_myria_node.sh every day at midnight
0 0 * * * /root/"node_name"-API/start_myria_node.sh > /root/Myria-API/node.log 2>&1

# Run start_myria_node_API2.sh every day at noon
0 12 * * * /root/"node_name"-API/start_myria_node_API2.sh > /root/Myria-API/node.log 2>&1

SendBATscript

Setting up BAT script for ez file transportation

#SendBAT script

User
@echo off
setlocal enabledelayedexpansion

set /p "localFile=Enter the local file path: "
set /p "password=Enter the remote user password: "
set "remoteUser=**"
set "remoteHost="***.***.*.**"
set "remotePath=(file/path at location)"

echo Copying "!localFile!" to "!remoteUser!@!remoteHost!:!remotePath!"

"C:\Program Files\PuTTY\pscp.exe" -pw !password! "!localFile!" "!remoteUser!@!remoteHost!:!remotePath!"

echo Copy completed.

pause
** = user and host data/IP's
